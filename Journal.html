// =============================== // Adjusting Journal Entries – Logic // =============================== const entriesTableBody = document.getElementById("entriesTableBody"); const statusButtons = document.querySelectorAll(".status-btn"); const entryDateFrom = document.getElementById("entryDateFrom"); const entryDateTo = document.getElementById("entryDateTo"); const entrySearch = document.getElementById("entrySearch"); let currentStatusFilter = "all"; // Render table rows based on current filters function renderAdjustingEntries() { // Clear current rows entriesTableBody.innerHTML = ""; const fromVal = entryDateFrom.value || null; const toVal = entryDateTo.value || null; const searchVal = entrySearch.value.trim().toLowerCase(); // Filter the data const filtered = adjustingEntries.filter(entry => { // Status filter if (currentStatusFilter !== "all" && entry.status !== currentStatusFilter) { return false; } // Date filter if (fromVal && entry.date < fromVal) { return false; } if (toVal && entry.date > toVal) { return false; } // Search filter (accounts, description, amount) if (searchVal) { const combinedText = `${entry.accounts} ${entry.description} ${entry.amount}`.toLowerCase(); if (!combinedText.includes(searchVal)) { return false; } } return true; }); // If no results if (filtered.length === 0) { const tr = document.createElement("tr"); const td = document.createElement("td"); td.colSpan = 5; td.textContent = "No adjusting entries match your filters."; td.style.textAlign = "center"; td.style.color = "#8890aa"; tr.appendChild(td); entriesTableBody.appendChild(tr); return; } // Build rows filtered.forEach(entry => { const tr = document.createElement("tr"); // Date const dateTd = document.createElement("td"); dateTd.textContent = entry.date; tr.appendChild(dateTd); // Accounts const accountsTd = document.createElement("td"); accountsTd.textContent = entry.accounts; tr.appendChild(accountsTd); // Description const descTd = document.createElement("td"); descTd.textContent = entry.description; tr.appendChild(descTd); // Amount const amountTd = document.createElement("td"); amountTd.textContent = entry.amount.toLocaleString("en-US", { minimumFractionDigits: 2, maximumFractionDigits: 2 }); tr.appendChild(amountTd); // Status const statusTd = document.createElement("td"); const span = document.createElement("span"); span.classList.add("status-pill"); if (entry.status === "pending") { span.classList.add("status-pending"); span.textContent = "Pending"; } else if (entry.status === "approved") { span.classList.add("status-approved"); span.textContent = "Approved"; } else if (entry.status === "rejected") { span.classList.add("status-rejected"); span.textContent = "Rejected"; } statusTd.appendChild(span); tr.appendChild(statusTd); entriesTableBody.appendChild(tr); }); } // Status button clicks statusButtons.forEach(btn => { btn.addEventListener("click", () => { statusButtons.forEach(b => b.classList.remove("active")); btn.classList.add("active"); currentStatusFilter = btn.getAttribute("data-status"); renderAdjustingEntries(); }); }); // Date & search filters entryDateFrom.addEventListener("change", renderAdjustingEntries); entryDateTo.addEventListener("change", renderAdjustingEntries); entrySearch.addEventListener("input", renderAdjustingEntries); // Initial render on page load renderAdjustingEntries();
